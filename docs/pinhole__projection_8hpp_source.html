43 template<typename Distortion>
44 void PinholeProjection<Distortion>::project3(
45  const Eigen::Ref<const Eigen::Vector3d>& point_3d,
46  Eigen::Vector2d* out_keypoint,
47  Eigen::Matrix<double, 2, 3>* out_jacobian_point) const
48 {
49  const double z_inv = 1 / point_3d(2);
50  // TODO(tcies) precalced member?
51  const Eigen::DiagonalMatrix<double, 2> focal_matrix(fx_, fy_);
52  const Eigen::Vector2d uv = point_3d.head<2>() * z_inv;
53  (*out_keypoint) =
54  focal_matrix * distortion_.distort(uv) + Eigen::Vector2d(cx_, cy_);
55 
56  if(out_jacobian_point)
57  {
58  Eigen::Matrix<double, 2, 3> duv_dxy;
59  duv_dxy.leftCols<2>() = Eigen::Matrix2d::Identity() * z_inv;
60  duv_dxy.rightCols<1>() = - point_3d.head<2>() * z_inv * z_inv;
61 
62  (*out_jacobian_point) = focal_matrix * distortion_.jacobian(uv) * duv_dxy;
63  }
64 }
