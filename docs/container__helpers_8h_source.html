13 template <typename Type>
14 void eraseIndicesFromVector(const std::vector<size_t>& sorted_indices_to_remove,
15  std::vector<Type>* vector)
16 {
17  CHECK_NOTNULL(vector);
18  std::vector<Type> to_swap;
19  to_swap.reserve(vector->size() - sorted_indices_to_remove.size());
20 
21  size_t in_begin = 0u;
22 
23  auto copy_until_and_skip = [&](const size_t end_index) {
24  CHECK_GE(end_index, in_begin) << "Indices to remove must be sorted!";
25  // TODO(tcies) Potential optimization: Use move iterators. Can't use memcpy
26  // because that will e.g. fail on shared pointers.
27  to_swap.insert(to_swap.end(), vector->begin() + in_begin,
28  vector->begin() + end_index);
29  in_begin = end_index + 1;
30  };
31 
32  // Copy blocks before indices to remove.
33  for (const size_t index_to_remove : sorted_indices_to_remove)
34  {
35  copy_until_and_skip(index_to_remove);
36  }
37 
38  // Copy block after last index to remove.
39  copy_until_and_skip(vector->size());
40 
41  vector->swap(to_swap);
42 }

45 // Solution from:
46 // http://stackoverflow.com/questions/7571937/how-to-delete-items-from-a-stdvector-given-a-list-of-indices
47 template<typename ElementType, typename Allocator>
48 inline std::vector<ElementType, Allocator> eraseIndicesFromVector_DEPRECATED(
49  const std::vector<ElementType, Allocator>& data,
50  const std::vector<size_t>& indices_to_delete) {
51  if (indices_to_delete.empty()) {
52  return data;
53  }
54  std::vector<size_t> mutable_indices_to_delete = indices_to_delete;
55  std::sort(mutable_indices_to_delete.begin(), mutable_indices_to_delete.end());
56  CHECK_LT(mutable_indices_to_delete.back(), data.size());
57 
58  std::vector<ElementType, Allocator> reduced_vector;
59  CHECK_GE(data.size(), mutable_indices_to_delete.size());
60  reduced_vector.reserve(data.size() - mutable_indices_to_delete.size());
61 
62  // Copy blocks from the input vector to the output vector.
63  typename std::vector<ElementType, Allocator>::const_iterator it_block_begin = data.begin();
64 
65  for (typename std::vector<size_t>::const_iterator it = mutable_indices_to_delete.begin();
66  it != mutable_indices_to_delete.end(); ++it) {
67  typename std::vector<ElementType, Allocator>::const_iterator it_block_end = data.begin() + *it;
68  if (it_block_begin != it_block_end) {
69  std::copy(it_block_begin, it_block_end, std::back_inserter(reduced_vector));
70  }
71  it_block_begin = it_block_end + 1;
72  }
73 
74  // Copy the last block.
75  if (it_block_begin != data.end()) {
76  std::copy(it_block_begin, data.end(), std::back_inserter(reduced_vector));
77  }
78  return reduced_vector;
79 }
