34  inline void distort(double& x, double& y) const
35  {
36  const double xx = x * x;
37  const double yy = y * y;
38  const double xy = x * y;
39  const double xy2 = 2.0*xy;
40  const double r2 = xx + yy;
41  const double cdist = (k1_ + k2_ * r2) * r2;
42  x += x * cdist + p1_ * xy2 + p2_ * (r2 + 2.0 * xx);
43  y += y * cdist + p2_ * xy2 + p1_ * (r2 + 2.0 * yy);
44  }

46  inline Eigen::Vector2d distort(const Eigen::Vector2d& vector) const {
47  const double xx = vector(0) * vector(0);
48  const double yy = vector(1) * vector(1);
49  const double xy = vector(0) * vector(1);
50  const double xy2 = 2.0*xy;
51  const double r2 = xx + yy;
52  const double cdist = (k1_ + k2_ * r2) * r2;
53  return Eigen::Vector2d(
54  vector(0) + vector(0) * cdist + p1_ * xy2 + p2_ * (r2 + 2.0 * xx),
55  vector(1) + vector(1) * cdist + p2_ * xy2 + p1_ * (r2 + 2.0 * yy));

59  inline Eigen::Matrix2d jacobian(const Eigen::Vector2d& px_distorted) const {
60  Eigen::Matrix2d J_dist;
61  const double xx = px_distorted[0] * px_distorted[0];
62  const double yy = px_distorted[1] * px_distorted[1];
63  const double xy = px_distorted[0] * px_distorted[1];
64  const double r2 = xx + yy;
65  const double cdist = (k1_ + k2_ * r2) * r2;
66 
67  const double k2_r2_x4 = k2_ * r2 * 4.0;
68  const double cdist_p1 = cdist + 1.0;
69  J_dist(0,0) = cdist_p1 + k1_ * 2.0 * xx + k2_r2_x4 * xx
70  + 2.0 * p1_ * px_distorted[1] + 6.0 * p2_ * px_distorted[0];
71  J_dist(1,1) = cdist_p1 + k1_ * 2.0 * yy + k2_r2_x4 * yy
72  + 2.0 * p2_ * px_distorted[0] + 6.0 * p1_ * px_distorted[1] ;
73  J_dist(1,0) = 2.0 * k1_ * xy + k2_r2_x4 * xy
74  + 2.0 * p1_ * px_distorted[0] + 2.0 * p2_ * px_distorted[1] ;
75  J_dist(0,1) = J_dist(1,0);
76  return J_dist;
77  }

79  inline void undistort(double& x, double& y) const
80  {
81  double x0=x, y0=y;
82  for(int i = 0; i < 5; ++i)
83  {
84  const double xx = x*x;
85  const double yy = y*y;
86  const double xy = x*y;
87  const double xy2 = 2*xy;
88  const double r2 = xx + yy;
89  const double icdist = 1.0/(1.0 + (k1_ + k2_ * r2) * r2);
90  const double dx = p1_ * xy2 + p2_ * (r2 + 2.0 * xx);
91  const double dy = p2_ * xy2 + p1_ * (r2 + 2.0 * yy);
92  x = (x0 - dx)*icdist;
93  y = (y0 - dy)*icdist;
94  }
95  }
