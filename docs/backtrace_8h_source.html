12 // Adapted from http://panthema.net/2008/0901-stacktrace-demangled/ .
13 inline std::string backtrace() {
14  constexpr size_t kBacktraceLimit = 64;
15  void* array[kBacktraceLimit];
16  size_t size;
17 
18  size = ::backtrace(array, kBacktraceLimit);
19  // Causes
20  // http://valgrind.org/docs/manual/mc-manual.html#opt.show-mismatched-frees ,
21  // but that is ok.
22  std::unique_ptr<char * []> symbollist(backtrace_symbols(array, size));
23 
24  if (size == 0u) {
25  return " Backtrace empty, possibly corrupt!";
26  }
27 
28  std::stringstream result;
29  for (size_t i = 1u; i < size; i++) {
30  char* begin_name = 0, *begin_offset = 0, *end_offset = 0;
31 
32  // Find parentheses and +address offset surrounding the mangled name:
33  // ./module(function+0x15c) [0x8048a6d]
34  for (char* p = symbollist[i]; *p; ++p) {
35  if (*p == '(') {
36  begin_name = p;
37  } else if (*p == '+') {
38  begin_offset = p;
39  } else if (*p == ')' && begin_offset) {
40  end_offset = p;
41  break;
42  }
43  }
44 
45  if (begin_name && begin_offset && end_offset && begin_name < begin_offset) {
46  *begin_name++ = '\0';
47  *begin_offset++ = '\0';
48  *end_offset = '\0';
49 
50  // Mangled name is now in [begin_name, begin_offset) and caller
51  // offset in [begin_offset, end_offset).
52  int status;
53  // Causes
54  // http://valgrind.org/docs/manual/mc-manual.html#opt.show-mismatched-frees
55  // but that is ok.
56  std::unique_ptr<char> demangled(
57  abi::__cxa_demangle(begin_name, nullptr, nullptr, &status));
58  if (status == 0) {
59  result << " " << symbollist[i] << " : " << demangled.get() << "+"
60  << begin_offset << "\n";
61  } else {
62  // Demangling failed. Output function name as a C function with
63  // no arguments.
64  result << " " << symbollist[i] << " : " << begin_name << "+"
65  << begin_offset << "\n";
66  }
67  } else {
68  // Couldn't parse the line? Print the whole line.
69  result << " " << symbollist[i] << "\n";
70  }
71  }
72 
73  return result.str();
74 }
