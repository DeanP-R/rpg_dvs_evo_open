42 void MiniLeastSquaresSolver<D, T, Implementation>::optimizeGaussNewton(State& state)
43 {
44  // Save the old model to rollback in case of unsuccessful update
45  State old_state = state;
46 
47  // perform iterative estimation
48  for (iter_ = 0; iter_<solver_options_.max_iter; ++iter_)
49  {
50  rho_ = 0;
51  startIteration();
52 
53  H_.setZero();
54  g_.setZero();
55 
56  // compute initial error
57  n_meas_ = 0;
58  double new_chi2 = evaluateError(state, &H_, &g_);
59 
60  // add prior
61  if(have_prior_)
62  {
63  applyPrior(state);
64  }
65 
66  // solve the linear system
67  if(!solve(H_, g_, dx_))
68  {
69  LOG(WARNING) << "Matrix is close to singular! Stop Optimizing."
70  << "H = " << H_ << "g = " << g_;
71  stop_ = true;
72  }
73 
74  // check if error increased since last optimization
75  if((iter_ > 0 && new_chi2 > chi2_ && solver_options_.stop_when_error_increases) || stop_)
76  {
77  VLOG(400) << "It. " << iter_
78  << "\t Failure"
79  << "\t new_chi2 = " << new_chi2
80  << "\t n_meas = " << n_meas_
81  << "\t Error increased. Stop optimizing.";
82  state = old_state; // rollback
83  break;
84  }
85 
86  // update the model
87  State new_state;
88  update(state, dx_, new_state);
89  old_state = state;
90  state = new_state;
91  chi2_ = new_chi2;
92  double x_norm = utils::norm_max(dx_);
93  VLOG(400) << "It. " << iter_
94  << "\t Success"
95  << "\t new_chi2 = " << new_chi2
96  << "\t n_meas = " << n_meas_
97  << "\t x_norm = " << x_norm;
98  finishIteration();
99 
100  // stop when converged, i.e. update step too small
101  if(x_norm < solver_options_.eps)
102  {
103  VLOG(400) << "Converged, x_norm " << x_norm << " < " << solver_options_.eps;
104  break;
105  }
106  }

109 template <int D, typename T, typename Implementation>
110 void MiniLeastSquaresSolver<D, T, Implementation>::optimizeLevenbergMarquardt(State& state)
111 {
112  // init parameters
113  mu_ = solver_options_.mu_init;
114  nu_ = solver_options_.nu_init;
115 
116  // compute the initial error
117  chi2_ = evaluateError(state, nullptr, nullptr);
118  VLOG(400) << "init chi2 = " << chi2_
119  << "\t n_meas = " << n_meas_;
120 
121  // TODO: compute initial lambda
122  // Hartley and Zisserman: "A typical init value of lambda is 10^-3 times the
123  // average of the diagonal elements of J'J"
124  // Compute Initial Lambda
125  if(mu_ < 0)
126  {
127  double H_max_diag = 0;
128  double tau = 1e-4;
129  for(size_t j=0; j<D; ++j)
130  {
131  H_max_diag = std::max(H_max_diag, std::fabs(H_(j,j)));
132  }
133  mu_ = tau*H_max_diag;
134  }
135 
136  // perform iterative estimation
137  for (iter_ = 0; iter_<solver_options_.max_iter; ++iter_)
138  {
139  rho_ = 0;
140  startIteration();
141 
142  // try to compute and update, if it fails, try with increased mu
143  trials_ = 0;
144  do
145  {
146  // init variables
147  State new_model;
148  double new_chi2 = -1;
149  H_.setZero();
150  //H_ = mu_ * Matrix<double,D,D>::Identity(D,D);
151  g_.setZero();
152 
153  // linearize
154  n_meas_ = 0;
155  evaluateError(state, &H_, &g_);
156 
157  // add damping term:
158  H_ += (H_.diagonal()*mu_).asDiagonal();
159 
160  // add prior
161  if(have_prior_)
162  {
163  applyPrior(state);
164  }
165 
166  // solve the linear system to obtain small perturbation in direction of gradient
167  if(solve(H_, g_, dx_))
168  {
169  // apply perturbation to the state
170  update(state, dx_, new_model);
171 
172  // compute error with new model and compare to old error
173  n_meas_ = 0;
174  new_chi2 = evaluateError(new_model, nullptr, nullptr);
175  rho_ = chi2_-new_chi2;
176  }
177  else
178  {
179  LOG(WARNING) << "Matrix is close to singular! Stop Optimizing."
180  << "H = " << H_ << "g = " << g_;
181  rho_ = -1;
182  }
183 
184  if(rho_>0)
185  {
186  // update decrased the error -> success
187  state = new_model;
188  chi2_ = new_chi2;
189  stop_ = utils::norm_max(dx_) < solver_options_.eps;
190  mu_ *= std::max(1./3., std::min(1.-std::pow(2*rho_-1,3), 2./3.));
191  nu_ = 2.;
192  VLOG(400) << "It. " << iter_
193  << "\t Trial " << trials_
194  << "\t Success"
195  << "\t n_meas = " << n_meas_
196  << "\t new_chi2 = " << new_chi2
197  << "\t mu = " << mu_
198  << "\t nu = " << nu_;
199  }
200  else
201  {
202  // update increased the error -> fail
203  mu_ *= nu_;
204  nu_ *= 2.;
205  ++trials_;
206  if (trials_ >= solver_options_.max_trials)
207  stop_ = true;
208 
209  VLOG(400) << "It. " << iter_
210  << "\t Trial " << trials_
211  << "\t Failure"
212  << "\t n_meas = " << n_meas_
213  << "\t new_chi2 = " << new_chi2
214  << "\t mu = " << mu_
215  << "\t nu = " << nu_;
216  }
217  finishTrial();
218 
219  } while(!(rho_>0 || stop_));
220  if (stop_)
221  {
222  break;
223  }
224 
225  finishIteration();
226  }
