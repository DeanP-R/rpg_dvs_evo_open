64  inline Eigen::Matrix2d jacobian(const Eigen::Vector2d& uv) const
65  {
66  const double r = uv.norm();
67  if (r < kRThresh)
68  {
69  return Eigen::Matrix2d::Identity();
70  }
71 
72  const double inv_r = 1.0 / r;
73  const double r2 = r * r;
74  const double dr_du = uv(0) * inv_r;
75  const double dr_dv = uv(1) * inv_r;
76 
77  const double theta = std::atan(r);
78  const double dtheta_dr = 1.0 / (1 + r * r);
79 
80  const double thetad = thetad_from_theta(theta);
81  const double dthetad_dtheta = deriv_thetad_from_theta(theta);
82  const double dthetad_dr = dthetad_dtheta * dtheta_dr;
83 
84  const double scaling = thetad / r;
85  const double dscaling_du = (dthetad_dr * dr_du * r - dr_du * thetad) / r2;
86  const double dscaling_dv = (dthetad_dr * dr_dv * r - dr_dv * thetad) / r2;
87 
88  const double dx_du = dscaling_du * uv(0) + scaling;
89  const double dx_dv = dscaling_dv * uv(0);
90  const double dy_du = dscaling_du * uv(1);
91  const double dy_dv = dscaling_dv * uv(1) + scaling;
92  Eigen::Matrix2d jac;
93  jac << dx_du, dx_dv, dy_du, dy_dv;
94  return jac;
95  }

97  inline void undistort(double& x, double& y) const
98  {
99  const double thetad = std::sqrt(x * x + y * y);
100  double theta = thetad;
101  for (int i = 0; i < 5; ++i)
102  {
103  const double theta2 = theta * theta;
104  const double theta4 = theta2 * theta2;
105  const double theta6 = theta4 * theta2;
106  const double theta8 = theta4 * theta4;
107  theta = thetad /
108  (1.0 + k1_ * theta2 + k2_ * theta4 + k3_ * theta6 + k4_ * theta8);
109  }
110  const double scaling = std::tan(theta) / thetad;
111  x *= scaling;
112  y *= scaling;
113  }
