36 inline bool readNextLine(
37  const size_t num_fields_per_line,
38  std::ifstream& csv_file_stream,
39  std::vector<std::string>* fields)
40 {
41  CHECK_NOTNULL(fields)->clear();
42  // Read the line.
43  std::string line;
44  if (!std::getline(csv_file_stream, line)) {
45  return false;
46  }
47 
48  // Treat empty lines as end-of-file (hopefully at the end of the file).
49  removeAnyEndOfLineCharacter(&line);
50  if (line.empty()) {
51  return false;
52  }
53 
54  // Split fields of the read line.
55  const char field_delimiter = ',';
56  fields->resize(num_fields_per_line);
57  std::stringstream line_stream(line);
58  for (size_t idx = 0u; idx < num_fields_per_line; ++idx) {
59  CHECK(!line_stream.eof());
60  std::getline(line_stream, (*fields)[idx], field_delimiter);
61 
62  CHECK(!(*fields)[idx].empty()) << "Too few fields per line."
63  << " Found " << idx + 1 << " fields but expected " << num_fields_per_line << ".";
64  }
65  return true;
66 }

112 inline Quaternion convertStringsToQuaternion(
113  const std::string& qw, const std::string& qx, const std::string& qy, const std::string& qz) {
114  CHECK(!qw.empty());
115  CHECK(!qx.empty());
116  CHECK(!qy.empty());
117  CHECK(!qz.empty());
118  Quaternion quaternion;
119  quaternion.setIdentity();
120  try {
121  constexpr double kNormTolerance = 1e-3;
122  Eigen::Vector4d quat_coeffs(std::stod(qw), std::stod(qx), std::stod(qy), std::stod(qz));
123  LOG_IF(FATAL, std::fabs(1.0 - quat_coeffs.norm()) > kNormTolerance)
124  << "Invalid quaternion with norm: " << quat_coeffs.norm();
125  quat_coeffs.normalize();
126  quaternion = Quaternion(quat_coeffs(0), quat_coeffs(1),
127  quat_coeffs(2), quat_coeffs(3));
128  } catch(const std::invalid_argument& invalid_argument_exception) {
129  LOG(FATAL) << "Invalid argument exception. Could not convert '(" << qw << ", "
130  << qx << ", " << qy << ", " << qz << ")' to aslam::Quaternion."
131  << invalid_argument_exception.what();
132  } catch(const std::out_of_range& out_of_range_exception) {
133  LOG(FATAL) << "Out of range exception. The given tuple '(" << qw << ", "
134  << qx << ", " << qy << ", " << qz << ")' does not seem to fit the definition "
135  << "of an aslam::Quaternion!" << out_of_range_exception.what();
136  } catch(const std::exception& exception) {
137  LOG(FATAL) << "Conversion of the given tuple '(" << qw << ", "
138  << qx << ", " << qy << ", " << qz << ")' to aslam::Quaternion "
139  << "failed. " << exception.what();
140  }
141  return quaternion;
142 }
