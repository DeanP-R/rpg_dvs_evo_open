242  inline static void jacobian_xyz2uv(
243  const Eigen::Vector3d& xyz_in_f,
244  Eigen::Matrix<double,2,6>& J)
245  {
246  const double x = xyz_in_f[0];
247  const double y = xyz_in_f[1];
248  const double z_inv = 1./xyz_in_f[2];
249  const double z_inv_2 = z_inv*z_inv;
250 
251  J(0,0) = -z_inv; // -1/z
252  J(0,1) = 0.0; // 0
253  J(0,2) = x*z_inv_2; // x/z^2
254  J(0,3) = y*J(0,2); // x*y/z^2
255  J(0,4) = -(1.0 + x*J(0,2)); // -(1.0 + x^2/z^2)
256  J(0,5) = y*z_inv; // y/z
257 
258  J(1,0) = 0.0; // 0
259  J(1,1) = -z_inv; // -1/z
260  J(1,2) = y*z_inv_2; // y/z^2
261  J(1,3) = 1.0 + y*J(1,2); // 1.0 + y^2/z^2
262  J(1,4) = -J(0,3); // -x*y/z^2
263  J(1,5) = -x*z_inv; // -x/z
264  }

267  inline static void jacobian_xyz2uv_imu(
268  const Transformation& T_cam_imu,
269  const Eigen::Vector3d& p_in_imu,
270  Eigen::Matrix<double,2,6>& J)
271  {
272  Eigen::Matrix<double,3,6> G_x; // Generators times pose
273  G_x.block<3,3>(0,0) = Eigen::Matrix3d::Identity();
274  G_x.block<3,3>(0,3) = -vk::skew(p_in_imu);
275  const Eigen::Vector3d p_in_cam = T_cam_imu * p_in_imu;
276 
277  Eigen::Matrix<double,2,3> J_proj; // projection derivative
278  J_proj << 1, 0, -p_in_cam[0]/p_in_cam[2],
279  0, 1, -p_in_cam[1]/p_in_cam[2];
280 
281  J = - 1.0/p_in_cam[2] * J_proj * T_cam_imu.getRotation().getRotationMatrix() * G_x;
282  }

285  inline static void jacobian_xyz2img_imu(
286  const Transformation& T_cam_imu,
287  const Eigen::Vector3d& p_in_imu,
288  const Eigen::Matrix<double, 2, 3>& J_cam,
289  Eigen::Matrix<double,2,6>& J)
290  {
291  Eigen::Matrix<double,3,6> G_x; // Generators times pose
292  G_x.block<3,3>(0,0) = Eigen::Matrix3d::Identity();
293  G_x.block<3,3>(0,3) = -vk::skew(p_in_imu);
294 
295  J = J_cam * T_cam_imu.getRotation().getRotationMatrix() * G_x;
296  }

299  inline static void jacobian_xyz2f_imu(
300  const Transformation& T_cam_imu,
301  const Eigen::Vector3d& p_in_imu,
302  Eigen::Matrix<double, 3, 6>& J)
303  {
304  Eigen::Matrix<double,3,6> G_x;
305  G_x.block<3,3>(0,0) = Eigen::Matrix3d::Identity();
306  G_x.block<3,3>(0,3) = -vk::skew(p_in_imu);
307  const Eigen::Vector3d p_in_cam = T_cam_imu * p_in_imu;
308 
309  Eigen::Matrix<double, 3, 3> J_normalize;
310  double x2 = p_in_cam[0]*p_in_cam[0];
311  double y2 = p_in_cam[1]*p_in_cam[1];
312  double z2 = p_in_cam[2]*p_in_cam[2];
313  double xy = p_in_cam[0]*p_in_cam[1];
314  double yz = p_in_cam[1]*p_in_cam[2];
315  double zx = p_in_cam[2]*p_in_cam[0];
316  J_normalize << y2+z2, -xy, -zx,
317  -xy, x2+z2, -yz,
318  -zx, -yz, x2+y2;
319  J_normalize *= 1 / std::pow(x2+y2+z2, 1.5);
320 
321  J = J_normalize * T_cam_imu.getRotationMatrix() * G_x;
322  }
