18 inline void createPatchFromPatchWithBorder(
19  const uint8_t* const patch_with_border,
20  const int patch_size,
21  uint8_t* patch)
22 {
23  uint8_t* patch_ptr = patch;
24  for(int y=1; y<patch_size+1; ++y, patch_ptr += patch_size)
25  {
26  const uint8_t* ref_patch_border_ptr = patch_with_border + y*(patch_size+2) + 1;
27  for(int x=0; x<patch_size; ++x)
28  patch_ptr[x] = ref_patch_border_ptr[x];
29  }
30 }

42 inline void normalizeAndUpsamplePatch(
43  const cv::Mat& patch,
44  double upsample_factor,
45  cv::Mat* img_rgb)
46 {
47  CHECK_NOTNULL(img_rgb);
48 
49  cv::Mat patch_normalized;
50  if(patch.type() == CV_32FC1)
51  {
52  double minval, maxval;
53  cv::minMaxLoc(patch, &minval, &maxval);
54  patch_normalized = (patch - minval) / (maxval - minval);
55  }
56  else if(patch.type() == CV_8UC1)
57  {
58  patch.convertTo(patch_normalized, CV_32FC1);
59  double minval, maxval;
60  cv::minMaxLoc(patch_normalized, &minval, &maxval);
61  patch_normalized = (patch_normalized - minval) / (maxval - minval);
62  }
63  else
64  {
65  LOG(FATAL) << "Image Type not supported.";
66  }
67  cv::Mat img_gray;
68  cv::resize(patch_normalized, img_gray, cv::Size(0,0), upsample_factor, upsample_factor, cv::INTER_NEAREST);
69  *img_rgb = cv::Mat(img_gray.size(), CV_8UC3);
70  cv::cvtColor(img_gray, *img_rgb, cv::COLOR_GRAY2RGB);
71 }

73 inline void concatenatePatches(
74  std::vector<cv::Mat> patches,
75  cv::Mat* result_rgb)
76 {
77  const size_t n = patches.size();
78  const int width = patches.at(0).cols;
79  const int height = patches.at(0).rows;
80 
81  *result_rgb = cv::Mat(height, n*width, CV_32FC3);
82  for(size_t i = 0; i < n; ++i)
83  {
84  CHECK_EQ(width, patches[i].cols);
85  CHECK_EQ(height, patches[i].rows);
86  cv::Mat roi(*result_rgb, cv::Rect(i*width, 0, width, height));
87  cv::Mat patch_rgb(patches[i].size(), CV_32FC3);
88  cv::cvtColor(patches[i], patch_rgb, cv::COLOR_GRAY2RGB);
89  patch_rgb.copyTo(roi);
90  }
91 }
