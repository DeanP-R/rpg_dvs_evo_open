158  inline static void jacobian_xyz2uv(
159  const Eigen::Vector3d& p_in_f,
160  const Eigen::Matrix3d& R_f_w,
161  svo::Matrix23d& point_jac)
162  {
163  const double z_inv = 1.0/p_in_f[2];
164  const double z_inv_sq = z_inv*z_inv;
165  point_jac(0, 0) = z_inv;
166  point_jac(0, 1) = 0.0;
167  point_jac(0, 2) = -p_in_f[0] * z_inv_sq;
168  point_jac(1, 0) = 0.0;
169  point_jac(1, 1) = z_inv;
170  point_jac(1, 2) = -p_in_f[1] * z_inv_sq;
171  point_jac = - point_jac * R_f_w;
172  }
173 
175  inline static void jacobian_xyz2f(
176  const Eigen::Vector3d& p_in_f,
177  const Eigen::Matrix3d& R_f_w,
178  Eigen::Matrix3d& point_jac)
179  {
180  Eigen::Matrix3d J_normalize;
181  double x2 = p_in_f[0]*p_in_f[0];
182  double y2 = p_in_f[1]*p_in_f[1];
183  double z2 = p_in_f[2]*p_in_f[2];
184  double xy = p_in_f[0]*p_in_f[1];
185  double yz = p_in_f[1]*p_in_f[2];
186  double zx = p_in_f[2]*p_in_f[0];
187  J_normalize << y2+z2, -xy, -zx,
188  -xy, x2+z2, -yz,
189  -zx, -yz, x2+y2;
190  J_normalize *= 1.0 / std::pow(x2+y2+z2, 1.5);
191  point_jac = (-1.0) * J_normalize * R_f_w;
192  }
193 };
