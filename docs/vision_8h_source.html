32 inline float
33 interpolateMat_32f(const cv::Mat& mat, float u, float v)
34 {
35  assert(mat.type()==CV_32F);
36  float x = floor(u);
37  float y = floor(v);
38  float subpix_x = u-x;
39  float subpix_y = v-y;
40  float wx0 = 1.0-subpix_x;
41  float wx1 = subpix_x;
42  float wy0 = 1.0-subpix_y;
43  float wy1 = subpix_y;
44 
45  float val00 = mat.at<float>(y,x);
46  float val10 = mat.at<float>(y,x+1);
47  float val01 = mat.at<float>(y+1,x);
48  float val11 = mat.at<float>(y+1,x+1);
49  return (wx0*wy0)*val00 + (wx1*wy0)*val10 + (wx0*wy1)*val01 + (wx1*wy1)*val11;
50 }

54 inline float
55 interpolateMat_8u(const cv::Mat& mat, float u, float v)
56 {
57  assert(mat.type()==CV_8U);
58  int x = floor(u);
59  int y = floor(v);
60  float subpix_x = u-x;
61  float subpix_y = v-y;
62 
63  float w00 = (1.0f-subpix_x)*(1.0f-subpix_y);
64  float w01 = (1.0f-subpix_x)*subpix_y;
65  float w10 = subpix_x*(1.0f-subpix_y);
66  float w11 = 1.0f - w00 - w01 - w10;
67 
68  const int stride = mat.step.p[0];
69  unsigned char* ptr = mat.data + y*stride + x;
70  return w00*ptr[0] + w01*ptr[stride] + w10*ptr[1] + w11*ptr[stride+1];
71 }
