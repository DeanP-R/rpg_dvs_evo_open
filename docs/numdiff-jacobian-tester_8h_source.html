126 class NumericalDiff : public _Functor {
127  public:
128  typedef _Functor Functor;
129  typedef typename Functor::Scalar Scalar;
130  typedef typename Functor::InputType InputType;
131  typedef typename Functor::ValueType ValueType;
132  typedef typename Functor::JacobianType JacobianType;

135  : Functor(f),
136  epsfcn(_epsfcn) {}
137 
138  enum {
139  InputsAtCompileTime = Functor::InputsAtCompileTime,
140  ValuesAtCompileTime = Functor::ValuesAtCompileTime
141  };

143  bool getJacobianNumerical(const InputType& _x, JacobianType &jac) const {
144  static_assert(InputsAtCompileTime > 0,
145  "Numerical Differentiation does not work for zero-sized Jacobians.");
146  using std::sqrt;
147  using std::abs;
148  /* Local variables */
149  Scalar h;
150  const typename InputType::Index n = _x.size();
151  const Scalar eps = sqrt(((std::max)(epsfcn, Eigen::NumTraits<Scalar>::epsilon())));
152 
153  // Build jacobian.
154  InputType x = _x;
155  ValueType val1, val2, val3, val4;
156  bool success = true; // Success of value generating function.
157 
158  for (int j = 0; j < n; ++j) {
159  h = eps * abs(x[j]);
160  if (h == 0.) {
161  h = eps;
162  }
163  switch (mode) {
164  case Forward:
165  success &= Functor::operator()(x, val1);
166  x[j] += h;
167  success &= Functor::operator()(x, val2);
168  x[j] = _x[j];
169  jac.col(j) = (val2 - val1) / h;
170  break;
171  case Central:
172  x[j] += h;
173  success &= Functor::operator()(x, val2);
174  x[j] -= 2 * h;
175  success &= Functor::operator()(x, val1);
176  x[j] = _x[j];
177  jac.col(j) = (val2 - val1) / (2 * h);
178  break;
179  case CentralSecond:
180  x[j] += 2.0 * h;
181  success &= Functor::operator()(x, val1);
182  x[j] -= h;
183  success &= Functor::operator()(x, val2);
184  x[j] -= 2.0 * h;
185  success &= Functor::operator()(x, val3);
186  x[j] -= h;
187  success &= Functor::operator()(x, val4);
188  x[j] = _x[j];
189  jac.col(j) = ((8.0 * val2) + val4 - val1 - (8.0 * val3)) / (h * 12.0);
190  break;
191  default:
192  eigen_assert(false);
193  };
194  }
195  return success;
196  }
197  private:
198  Scalar epsfcn;
199  NumericalDiff& operator=(const NumericalDiff&);
200 };
