113 // discrete Gradient operator on a uniform grid
114 // given u(x,y), it returns DX = a_1 du/dx, DY = a_2 du/dy,
115 // using one-sided finite differences
116 // U[i][j] = u(x_i, y_j), i=0...n-1
117 // DX[i][j] = a1*(U[i+1][j]-U[i][j])/h, i=0...n-2

119  const boost::multi_array<double,2>& U,
120  double a1, double a2, double h1, double h2);
121 // same as above, but the input U is assumed to contain only inner grid points
122 // and the boundary grid points are inferred from the boundary condition
123 // U[i][j] = u(x_i, y_j), i=0...n-1

145 // discrete Laplace operator on a uniform grid

187 // solves the 2D Poisson equation: a1 u_xx + a2 u_yy = f
188 //
189 // discretised with uniform grid, x_{i+1} = x_i + h1, y_{i+1} = y_i + h2
190 // input: rhs F[i][j] = f(x_i, y_j)
191 // output: solution U[i][j] = u(x_i, y_j), ie only inner points by default
192 //
193 // boundary condition:
194 // - lower and upper boundary in dimension 1: bd1a, bd1b
195 // - lower and upper boundary in dimension 2: bd2a, bd2b
196 // - e.g. bd1a[i] refers to U[-1][i]
